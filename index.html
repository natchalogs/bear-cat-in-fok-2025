        // --- RENDER CHECKLIST (UPDATED WITH EDIT) ---

        function renderChecklist(checklist) {

            const list = document.getElementById('checklist-list');

            const noChecklist = document.getElementById('no-checklist');

            if (!list || !noChecklist) return;

            list.innerHTML = '';

           

            const sortedList = (checklist || []).sort((a, b) => a.done - b.done);

            noChecklist.classList.toggle('hidden', sortedList.length > 0);

            sortedList.forEach(item => {

                const isDone = item.done;

                const isEditing = editingState.checklist === item.id;

                const createdByHtml = item.createdBy && !isEditing

                        ? `<span class="text-xs text-gray-400 italic ml-2">(โดย: ${item.createdBy.split(' ')[0]})</span>`

                        : '';

                if (isEditing) {

                    // EDIT MODE

                    list.innerHTML += `

                    <li class="flex gap-2 items-center p-2 bg-gray-50 rounded-md">

                        <input type="text" id="checklist-edit-input-${item.id}" value="${item.name}"

                               class="flex-grow p-1 border border-blue-300 rounded focus:outline-none focus:ring-1 focus:ring-blue-500">

                        <button onclick="window.app.updateChecklistItem('${item.id}')" class="text-green-600 hover:text-green-800 font-medium text-sm">บันทึก</button>

                        <button onclick="window.app.cancelChecklistEdit('${item.id}')" class="text-gray-500 hover:text-gray-700 text-sm">ยกเลิก</button>

                    </li>

                    `;

                } else {

                    // DISPLAY MODE

                    list.innerHTML += `

                    <li class="flex justify-between items-center p-2 hover:bg-gray-50 rounded-md group">

                        <div class="flex items-center flex-grow mr-2">

                            <span class="${isDone ? 'text-gray-500' : 'text-gray-800'}">${item.name}</span>

                            ${createdByHtml}

                            <button onclick="window.app.editChecklistItem('${item.id}')"

                                class="ml-2 opacity-0 group-hover:opacity-100 text-gray-400 hover:text-blue-500 transition-opacity" title="แก้ไข">

                                ✏️

                            </button>

                        </div>

                        <button onclick="window.app.toggleChecklistItem('${item.id}', ${!isDone})"

                                class="px-2 py-1 text-sm text-white rounded-md flex-shrink-0

                                ${isDone

                                    ? 'bg-gray-500 hover:bg-gray-600'

                                    : 'bg-green-500 hover:bg-green-600'

                                }">

                            ${isDone ? 'ยังไม่ทำ' : 'ทำแล้ว'}

                        </button>

                    </li>

                    `;

                }

            });

        }

        async function addDailyItem(dayIndex) {

            const timeInput = document.getElementById(`item-time-${dayIndex}`);

            const textInput = document.getElementById(`item-text-${dayIndex}`);

            if (!timeInput || !textInput || !appData) return;

            const time = timeInput.value;

            const text = textInput.value.trim();

            if (!text) {

                alert("กรุณาใส่รายละเอียดกิจกรรม");

                return;

            }

            const newItem = {

                id: `item${Date.now()}`,

                createdBy: userDisplayName

            };

            const newItinerary = appData.dailyItinerary.map((day, index) => {

                if (index === dayIndex) {

                    const newItems = (day.items || []).concat(newItem);

                    return { ...day, items: newItems };

                }

                return day;

            });

           

            await updateItinerary(newItinerary);

            textInput.value = '';

        }

        function autoFillPoiName() {

            const linkInput = document.getElementById('poi-link');

            const nameInput = document.getElementById('poi-name');

            const url = linkInput.value.trim();

            if (!url) {

                alert("กรุณาวางลิงก์ Google Maps ก่อนค่ะ");

                return;

            }

            // Simple Regex for permanent links

            const regex = /place\/([^\/]+)/;

            const match = url.match(regex);

            if (match && match[1]) {

                let extractedName = decodeURIComponent(match[1].replace(/\+/g, ' '));

                nameInput.value = extractedName;

                nameInput.classList.add('bg-green-100');

                setTimeout(() => nameInput.classList.remove('bg-green-100'), 1000);

            } else {

                alert("ไม่สามารถดึงชื่อจากลิงก์นี้ได้อัตโนมัติ (กรุณาพิมพ์ชื่อเองนะคะ)");

            }
